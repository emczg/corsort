[docs]    def next_compare(self):
        while True:
            chains = greedy_chain_decomposition(self.leq_)
            if len(chains) <= 1:
                break
            i_in_chain = 0
            j_in_chain = 0
            found = False
            while not found:
                i = chains[-1][i_in_chain]
                j = chains[-2][j_in_chain]
                if self.leq_[i, j] == 0:
                    found = True
                    yield i, j
                elif self.leq_[i, j] == 1:
                    i_in_chain += 1
                else:
                    j_in_chain += 1
